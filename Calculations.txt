Pet DPS = AutoDPS + ClawDPS
	ClawDPS = ClawsPerSecWOBite * ClawDMG
		ClawDMG = ClawCritDmg + ClawHitDmg
			ClawCritDmg = 200% * ClawCritChance * ClawAvgDmg * BossArmorReduction
				ClawCritChance = ((AgiFromBuffsAndGear / 20) + CritFromBuffsAndGear + (Ferocity * 3))  - 0.03 - 0.018 + BeastMastersGloves[2%]
				ClawAvgDmg = ClawAvg * PetHappy * PetFamilyMod * UnleashFuryMod * BeastWrath * OrcCommandMod * BMGearMod * AQGearMod
					ClawAvg = (43 + 59) / 2 [Rank 8 Claw]
					PetHappy = 1.25
					PetFamilyMod = Pet Reference Table.DmgMod [Cat = 10%]
					UnleashFuryMod = 1 + (0.04 * UnleashFuryPoints)
					BeastWrath = If BestialWrathTalent Then (BWUptime / FightDuration * 0.5 + 1.0) Else 1.0
						BWUptime = 18s every 2 mins (calc based on fight duration)
					OrcCommandMod = Orcs get 5% bonus
					BMGearMod = Beastmasters Boots give 3%
					AQGearMod = Unseen Path Set Bonus of 3% pet dmg
				BossArmorReduction = See sheet with armor reducing abilities - 98.32% in the default case (assumes sunders for 5%, expose for 95%)
			ClawHitDmg = ClawHitChance * ClawAvgDmg * BossArmorReduction
				ClawHitChance = 100% - ClawCritChance - ClawMissChance - ClawDodgeChance
					ClawMissChance = 8%
					ClawDodgeChance = 6.5%
		ClawsPerSecWOBite = NumClawsWOBite / TotalTime
			TotalTime = FightDuration * Uptime
			NumClawsWOBite = FocusGained / 25
				FocusGained = 100 + FocusPerSec * TotalTime
					FocusPerSec = FocusRegenAmount / FocusRegenTime
						FocusRegenAmount = (24 + 27) / 2
						FocusRegenTime = (4.8 + 5.6) / 2
	AutoDPS = AutoPerSec * AutoDMG
		AutoPerSec = AutoPerFight / TotalTime
			AutoPerFight = TotalTime / PetSpeed
		AutoDMG = AutoCritDmg + AutoHitDmg + AutoGlanceDmg
			AutoGlanceDmg = GlanceDmgPercent * GlanceChance * AutoAvgDamage * BossArmorReduction
				GlanceDmgPercent = 65%
				GlanceChance = 40%
				AutoAvgDamage = ((BaseDPS + PetAPMod) * PetSpeed) * PetHappy * PetFamilyMod * UnleashFuryMod * BeastWrath * OrcCommandMod * BMGearMod * AQGearMod + GiftofArthas
					GiftOfArthas = If procced +8 dmg
					BaseDPS = TooltipDPS - (BaseAP / 14)
						BaseAP = 252
						TooltipDPS = 40.944
					PetAPMod = PetAP / 14
						PetAP = (StrengthFromBuffsAndGear - 10) * 2 + APFromBuffsAndGear
							StrengthFromBuffsAndGear = RoundDown((136 + StrFromPetConsumes + 12or16 from MotW + EarthTotem) * KingsModPet * ZGModPet)

								ZGModPet = 1.1 if ZG is on pet
